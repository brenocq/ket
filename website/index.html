A Bell state in a few lines.

Build a circuit, simulate it, read the amplitudes — from C++, Python, or QASM.

#include <ket/ket.hpp>

int main() {
  ket::Circuit c{2};
  c.h(0);
  c.cx(0, 1);
  std::cout << ket::run(c).print();
}
import ket

c = ket.Circuit(2)
c.h(0)
c.cx(0, 1)
print(ket.run(c))
